function isPrime(number) {
  if (number <= 1) {
    return false; // Numbers less than or equal to 1 are not prime.
  }
  
  }
  for (let i = 5; i * i <= number; i += 6) {
    if (number % i === 0 || number % (i + 2) === 0) {
      return false; // Numbers divisible by i or i + 2 are not prime.
    }
  }
  return true;
}
